40  // loads a model with supported ASSIMP extensions from file and stores the resulting meshes in the meshes vector.
41  void loadModel(string const &path) {
42  // read file via ASSIMP
43  Assimp::Importer importer;
44  const aiScene* scene = importer.ReadFile(path, aiProcess_Triangulate | aiProcess_FlipUVs | aiProcess_CalcTangentSpace);
45  // check for errors
46  if(!scene || scene->mFlags & AI_SCENE_FLAGS_INCOMPLETE || !scene->mRootNode) {
47  cout << "ERROR::ASSIMP:: " << importer.GetErrorString() << endl;
48  return;
49  }
50  // retrieve the directory path of the filepath
51  directory = path.substr(0, path.find_last_of('/'));
52 
53  // process ASSIMP's root node recursively
54  processNode(scene->mRootNode, scene);
55  }
56  // processes a node in a recursive fashion. Processes each individual mesh located at the node and repeats this process on its children nodes (if any).
57  void processNode(aiNode *node, const aiScene *scene) {
58  // process each mesh located at the current node
59  for(unsigned int i = 0; i < node->mNumMeshes; i++) {
60  // the node object only contains indices to index the actual objects in the scene.
61  // the scene contains all the data, node is just to keep stuff organized (like relations between nodes).
62  aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
63  std::cout << "meshes push_back" << std::endl;
64  meshes.push_back(processMesh(mesh, scene));
65  }
66  // after we've processed all of the meshes (if any) we then recursively process each of the children nodes
67  for(unsigned int i = 0; i < node->mNumChildren; i++) {
68  processNode(node->mChildren[i], scene);
69  }
70  }

72  Mesh processMesh(aiMesh *mesh, const aiScene *scene) {
73  // data to fill
74  vector<Vertex> vertices;
75  vector<unsigned int> indices;
76  vector<Texture> textures;
77 
78  // Walk through each of the mesh's vertices
79  for(unsigned int i = 0; i < mesh->mNumVertices; i++) {
80  Vertex vertex;
81  glm::vec3 vector; // we declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
82  // positions
83  vector.x = mesh->mVertices[i].x;
84  vector.y = mesh->mVertices[i].y;
85  vector.z = mesh->mVertices[i].z;
86  vertex.Position = vector;
87  // normals
88  vector.x = mesh->mNormals[i].x;
89  vector.y = mesh->mNormals[i].y;
90  vector.z = mesh->mNormals[i].z;
91  vertex.Normal = vector;
92  // texture coordinates
93  if(mesh->mTextureCoords[0]) {
94  glm::vec2 vec;
95  // a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
96  // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
97  vec.x = mesh->mTextureCoords[0][i].x;
98  vec.y = mesh->mTextureCoords[0][i].y;
99  vertex.TexCoords = vec;
100  }
101  else {
102  vertex.TexCoords = glm::vec2(0.0f, 0.0f);
103  }
104  // tangent
105  vector.x = mesh->mTangents[i].x;
106  vector.y = mesh->mTangents[i].y;
107  vector.z = mesh->mTangents[i].z;
108  vertex.Tangent = vector;
109  // bitangent
110  vector.x = mesh->mBitangents[i].x;
111  vector.y = mesh->mBitangents[i].y;
112  vector.z = mesh->mBitangents[i].z;
113  vertex.Bitangent = vector;
114  vertices.push_back(vertex);
115  }
116  // now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
117  for(unsigned int i = 0; i < mesh->mNumFaces; i++) {
118  aiFace face = mesh->mFaces[i];
119  // retrieve all indices of the face and store them in the indices vector
120  for(unsigned int j = 0; j < face.mNumIndices; j++)
121  indices.push_back(face.mIndices[j]);
122  }
123  // process materials
124  aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
125  // we assume a convention for sampler names in the shaders. Each diffuse texture should be named
126  // as 'texture_diffuseN' where N is a sequential number ranging from 1 to MAX_SAMPLER_NUMBER.
127  // Same applies to other texture as the following list summarizes:
128  // diffuse: texture_diffuseN
129  // specular: texture_specularN
130  // normal: texture_normalN
131 
132  // 1. diffuse maps
133  vector<Texture> diffuseMaps = loadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
134  textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
135  // 2. specular maps
136  vector<Texture> specularMaps = loadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
137  textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
138  // 3. normal maps
139  std::vector<Texture> normalMaps = loadMaterialTextures(material, aiTextureType_HEIGHT, "texture_normal");
140  textures.insert(textures.end(), normalMaps.begin(), normalMaps.end());
141  // 4. height maps
142  std::vector<Texture> heightMaps = loadMaterialTextures(material, aiTextureType_AMBIENT, "texture_height");
143  textures.insert(textures.end(), heightMaps.begin(), heightMaps.end());
144 
145  // return a mesh object created from the extracted mesh data
146  return Mesh(vertices, indices, textures);
147  }

149  // checks all material textures of a given type and loads the textures if they're not loaded yet.
150  // the required info is returned as a Texture struct.
151  vector<Texture> loadMaterialTextures(aiMaterial *mat, aiTextureType type, string typeName)
152  {
153  vector<Texture> textures;
154  for(unsigned int i = 0; i < mat->GetTextureCount(type); i++) {
155  aiString str;
156  mat->GetTexture(type, i, &str);
157  // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
158  bool skip = false;
159  for(unsigned int j = 0; j < textures_loaded.size(); j++) {
160  if(std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0)
161  {
162  textures.push_back(textures_loaded[j]);
163  skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
164  break;
165  }
166  }
167  if(!skip) {
168  Texture texture;
169  texture.id = TextureFromFile(str.C_Str(), this->directory);
170  texture.type = typeName;
171  texture.path = str.C_Str();
172  textures.push_back(texture);
173  textures_loaded.push_back(texture); // store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
174  }
175  }
176  return textures;
177  }

179  unsigned int TextureFromFile(const char *path, const string &directory) {
180  string filename = string(path);
181  filename = directory + '/' + filename;
182 
183  unsigned int textureID;
184  glGenTextures(1, &textureID);
185 
186  int width, height, nrComponents;
187  unsigned char *data = stbi_load(filename.c_str(), &width, &height, &nrComponents, 0);
188  if (data)
189  {
190  GLenum format = 0;
191  if (nrComponents == 1)
192  format = GL_RED;
193  else if (nrComponents == 3)
194  format = GL_RGB;
195  else if (nrComponents == 4)
196  format = GL_RGBA;
197 
198  glBindTexture(GL_TEXTURE_2D, textureID);
199  glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
200  glGenerateMipmap(GL_TEXTURE_2D);
201 
202  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
203  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
204  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
205  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
206 
207  stbi_image_free(data);
208  }
209  else
210  {
211  std::cout << "Texture failed to load at path: " << path << std::endl;
212  stbi_image_free(data);
213  }
214  return textureID;
215  }
